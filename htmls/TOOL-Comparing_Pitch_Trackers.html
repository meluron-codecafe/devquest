Comparing Pitch Trackers

praat-parselmouth; pyreaper; librosa; pyworld; libtsm;

AIM¶

  1. To report different pitch trackers' performance in their default settings.
  2. To write easy-to-understand description for parameters used in these trackers.
  3. To adjust their parameters to get a better pitch contour.

Uncontrolled Test¶

We first try to test how different pitch trackers perform in their default settings. It is an uncontrolled experiment where we take an example audio without its ground truth f0 contour for comparison. The idea is to make some visual observations.

import parselmouth as pm
import pyworld as pw
import pyreaper as pr
import librosa as lb

import numpy as np
import modusa as ms
import time

# Load a sample audio signal
y, sr, title = ms.load("../../../data/fsm-snd/Crocodile 1.mp3", sr=16000)
ann = ms.load_ann("../../../data/fsm-snd/Crocodile 1-words.txt") # Loading word-level annotation (manually annotated)


def get_f0_contour(y, sr, lib):
    """
    Compute the F0 contour using different
    python libraries in their default
    settings.

    Parameters
    ----------
    y: ndarray
        Audio signal to compute the F0 contour of.
    sr: int
        Sampling rate of the audio signal.
    lib: str
        Which library to use for computing F0 contour.
        "pm" for parselmouth, "pw" for pyworld, "lr" for librosa, "pr" for pyreaper

    Returns
    -------
    ndarray
        F0 contour.
    ndarray
        time stamps.
    """

    # Parselmouth
    def f0_pm():
        snd = pm.Sound(values=y, sampling_frequency=sr)
        pitch_obj = snd.to_pitch_ac()
        f0 = pitch_obj.selected_array['frequency']
        t = pitch_obj.ts()

        return f0, t

    # Pyworld
    def f0_pw():
        _f0, t = pw.dio(y, sr)
        f0 = pw.stonemask(y, _f0, t, sr) # This is for refining pitch

        return f0, t

    # Librosa
    def f0_lr():
        f0, _, _ = lb.pyin(y=y, sr=sr, fmin=lb.note_to_hz('C2'), fmax=lb.note_to_hz('C7')) # Recommended parameter choice (check docstr)
        t = lb.times_like(f0, sr=sr)

        return f0, t

    # Pyreaper
    def f0_pr():
        y_int = (y * 32767).astype(np.int16)
        _, _, t, f0, _ = pr.reaper(x=y_int, fs=sr)

        return f0, t


    dispatcher = {
        "pm": f0_pm,
        "pw": f0_pw,
        "lr": f0_lr,
        "pr": f0_pr
    }

    if lib not in dispatcher:
        raise ValueError(f"Invalid lib keyword, choose from {dispatcher.keys}")

    # Print the time taken
    tic = time.time()
    f0, f0_t = dispatcher[lib]()
    toc = time.time()
    runtime = toc - tic # Seconds
    
    return f0, f0_t, runtime

# To better observe how these pitch trackers perform, we also look at the spectrogram
N, H = 1024, 160 # 64 ms, 10 ms
S = lb.stft(y, n_fft=N, win_length=N, hop_length=H)
S = np.log(1 + 10*(np.abs(S)**2))
S_f = np.arange(S.shape[0]) * (sr / N)
S_t = np.arange(S.shape[1]) * (H / sr)
