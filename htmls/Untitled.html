import matplotlib.pyplot as plt

def add_labeled_arrow(ax, xy, label, 
                      text_offset=(0, 0), 
                      arrowprops=None, 
                      bbox=None,
                      fontsize=12, 
                      c="black"):
    """
    Add an arrow pointing to a point with a label at the tail inside a box.

    Parameters
    ----------
    ax : matplotlib.axes.Axes
        Axis to draw on.
    xy : tuple[float, float]
        Target point (x, y) for the arrow head.
    label : str
        Text label at the arrow tail.
    text_offset : tuple[float, float]
        Offset (dx, dy) for label position from arrow tail.
    arrowprops : dict | None
        Arrow style.
    bbox : dict | None
        Bounding box style for the text label.
    fontsize : int
        Font size of the label.
    c : str
        Color for arrow and text.
    """
    if arrowprops is None:
        arrowprops = dict(arrowstyle="->", color=c, lw=1.2)
    if bbox is None:
        bbox = dict(boxstyle="round,pad=0.3", fc="white", ec=c, lw=1)

    # Tail position = shifted from head by text_offset
    text_x, text_y = xy[0] + text_offset[0], xy[1] + text_offset[1]

    ax.annotate(
        label,
        xy=xy, xycoords="data",
        xytext=(text_x, text_y), textcoords="data",
        arrowprops=arrowprops,
        fontsize=fontsize,
        color=c,
        ha="center", va="center",
        bbox=bbox
    )
